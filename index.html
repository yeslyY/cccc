import random # توليد رقم عشوائي بين 1 و 10 secret_number = random.randint(1, 10) print("🎲 اهلا بك في لعبة تخمين الرقم!") print("حاول تخمين الرقم بين 1 و 10") # المحاولات for attempt in range(3): guess = int(input("👉 ادخل تخمينك: ")) if guess == secret_number: print("🎉 مبروك! خمنت الرقم الصحيح 🎉") break elif guess < secret_number: print("📉 الرقم اكبر من تخمينك") else: print("📈 الرقم اصغر من تخمينك") else: print(f"😢 خسرت! الرقم الصحيح كان {secret_number}")
